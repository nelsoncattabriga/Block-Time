<objective>
Implement the one-time Core Data → SwiftData migration service (FOUND-09, FOUND-10, FOUND-11) as defined by user decisions D-06 through D-09. Three components:
1. `CoreDataMigrationService` (the orchestrator, state machine, flag management, row verification, `exit(0)` trigger).
2. `CoreDataMigrationActor` (`@ModelActor` for background writes — FOUND-11).
3. `LegacyFlightSnapshot` (Sendable DTO that crosses the actor boundary).

The service reads from v1's existing `FlightDatabaseService.shared.viewContext` (read-only — v1 stack untouched), converts via `TimeStringConverter` (Plan 02), and writes to a CloudKit-disabled `ModelContainer` from Plan 03. After row-count verification, it sets `migrationComplete=true` and calls `exit(0)` to force a clean relaunch into the CloudKit-enabled production container.

Purpose: This is the only thing standing between users and data loss. The service must be crash-safe (clear-and-retry per D-06), atomic on the flag write order (D-07), and verified by exact row count match (D-08).
Output: A migration service with three integration tests passing against a real v1 `.sqlite` fixture, plus crash-recovery and background-thread verification tests.
</objective>

    import Foundation
    import CoreData
    import SwiftData
    import os

    /// One-time Core Data → SwiftData migration orchestrator (FOUND-09).
    /// Reads v1 Core Data store (read-only, via existing FlightDatabaseService.shared.viewContext),
    /// writes to a CloudKit-disabled SwiftData ModelContainer (D-09), verifies row counts (D-08),
    /// and triggers `exit(0)` to force a clean relaunch into the CloudKit-enabled production container.
    ///
    /// Crash safety: clear-and-retry (D-06). Two UserDefaults flags, set in strict order (D-07).
    @MainActor
    final class CoreDataMigrationService {

        // MARK: - State machine

        enum State: Equatable {
            case notStarted
            case crashed       // started=true, complete=false
            case complete
        }

        // MARK: - Configuration

        /// Override hooks for testability.
        struct Dependencies {
            var defaults: UserDefaults = .standard
            var swiftDataStoreURL: URL = ModelContainerFactory.appGroupStoreURL()
            var makeMigrationContainer: () throws -> ModelContainer = { try ModelContainerFactory.makeMigrationContainer() }

        // MARK: - State

        var state: State {
            let started = deps.defaults.bool(forKey: Self.startedKey)
            let complete = deps.defaults.bool(forKey: Self.completeKey)
            switch (started, complete) {
            case (true, true):  return .complete
            case (true, false): return .crashed
            case (false, _):    return .notStarted
            }
        }

        // MARK: - Entry point

        /// Runs migration if necessary. Idempotent: safe to call on every launch.
        /// Must be called from @MainActor (e.g., SplashScreenView.task).
        func runIfNeeded() async throws {
            switch state {
            case .complete:
                logger.info("Migration already complete — skipping.")
                return
            case .crashed:
                logger.warning("Detected crashed migration. Deleting partial SwiftData store and retrying (D-06).")
                deletePartialStore()
                // Reset started flag; we are starting over.
                deps.defaults.set(false, forKey: Self.startedKey)
                fallthrough
            case .notStarted:
                try await performMigration()
            }
        }

        // MARK: - Private

        private func performMigration() async throws {
            // D-07: set started FIRST, before any work.
            deps.defaults.set(true, forKey: Self.startedKey)

            // Read Core Data on main thread (NSManagedObject is not Sendable).
            let snapshots: [LegacyFlightSnapshot]
            do {
                snapshots = try deps.fetchLegacySnapshots()
            } catch {
                throw MigrationError.coreDataReadFailed(underlying: error)
            }

            let sourceCount: Int
            do {
                sourceCount = try deps.coreDataCount()
            } catch {
                throw MigrationError.coreDataReadFailed(underlying: error)
            }

            logger.info("Migration: read \(snapshots.count, privacy: .public) snapshots from Core Data (count check: \(sourceCount, privacy: .public)).")

            // Build migration container with cloudKitDatabase: .none (D-09).
            let container: ModelContainer
            do {
                container = try deps.makeMigrationContainer()
            } catch {
                throw MigrationError.containerCreationFailed(underlying: error)
            }

            // Run @ModelActor write off the main thread (FOUND-11, RESEARCH.md §Pitfall 4).
            let writtenCount: Int = try await Task.detached(priority: .userInitiated) {
                let actor = CoreDataMigrationActor(modelContainer: container)
                do {
                    return try await actor.importLegacyFlights(snapshots)
                } catch {
                    throw MigrationError.swiftDataWriteFailed(underlying: error)
                }
            }.value

            // D-08: Verify row counts BEFORE setting complete.
            let destCount: Int = try await Task.detached(priority: .userInitiated) {
                let actor = CoreDataMigrationActor(modelContainer: container)
                return try await actor.count()
            }.value

            guard writtenCount == sourceCount, destCount == sourceCount else {
                logger.error("Row count mismatch — expected \(sourceCount, privacy: .public), wrote \(writtenCount, privacy: .public), dest \(destCount, privacy: .public). NOT setting complete flag.")
                throw MigrationError.rowCountMismatch(expected: sourceCount, actual: destCount)
            }

            // D-07: set complete LAST.
            deps.defaults.set(true, forKey: Self.completeKey)
            logger.info("Migration complete. \(destCount, privacy: .public) records. Triggering relaunch (D-09).")

            // D-09: force relaunch so the next launch uses the CloudKit-enabled container.
            deps.onComplete()
        }

        /// Deletes the SwiftData store file + sidecars (-shm, -wal) at the configured URL.
        /// Called on `.crashed` state per D-06.
        private func deletePartialStore() {
            let fm = FileManager.default
            let url = deps.swiftDataStoreURL
            let candidates = [url, url.appendingPathExtension("shm"), url.appendingPathExtension("wal")]
            for c in candidates {
                if fm.fileExists(atPath: c.path) {
                    do {
                        try fm.removeItem(at: c)
                        logger.info("Deleted partial store file: \(c.lastPathComponent, privacy: .public)")
                    } catch {
                        logger.error("Failed to delete partial store \(c.lastPathComponent, privacy: .public): \(error.localizedDescription, privacy: .public)")
                    }
                }
            }
        }
