struct DashboardCardID: RawRepresentable, Codable, Hashable, Identifiable {
    let rawValue: String
    var id: String { rawValue }
    init(rawValue: String) { self.rawValue = rawValue }

    // Every existing case becomes a static let with identical rawValue:
    static let frmsStatus = DashboardCardID(rawValue: "frmsStatus")
    static let customCount = DashboardCardID(rawValue: "customCount")
    // ... (preserve every existing case's rawValue exactly)

    // Custom counter factory uses prefix to keep parsing simple:
    static func customCounter(_ id: UUID) -> DashboardCardID {
        DashboardCardID(rawValue: "customCounter.\(id.uuidString)")
    }

    var customCounterID: UUID? {
        guard rawValue.hasPrefix("customCounter.") else { return nil }
        return UUID(uuidString: String(rawValue.dropFirst("customCounter.".count)))
